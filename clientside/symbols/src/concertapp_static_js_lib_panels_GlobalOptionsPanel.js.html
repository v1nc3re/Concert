  7 /**
  8  *  The panel on the top of every page.
  9  *	@class
 10  **/
 11 function GlobalOptionsPanel(params) {
 12     if(params) {
 13         this.init(params);
 14     }
 15 }
 16 GlobalOptionsPanel.prototype = new Panel();
 17 
 18 /**
 19  *  Initialize panel
 20  *
 21  *  @param  params.collectionSelector        jQuery HTMLSelectElement object - The
 22  *                                          select which chooses which collection to
 23  *                                          organize.
 24  *  @param  params.uploadButton             An UploadLinkLargeIconButton object.      
 25  *
 26  **/
 27 GlobalOptionsPanel.prototype.init = function(params) {
 28     Panel.prototype.init.call(this, params);
 29 
 30     var contents = this.contents;
 31     
 32     /* Make sure parameter was passed in */
 33     var collectionSelector = contents.find('.collection_selector');
 34     if(typeof(collectionSelector) == 'undefined' || collectionSelector.length == 0) {
 35         throw new Error('collectionSelector not found');
 36     }
 37     /* Save as member */
 38     this.collectionSelector = collectionSelector;
 39     
 40 
 41     /* Get upload button */
 42     var uploadButtonContainer = contents.find('#upload_button');
 43     /* If this page has an upload button */
 44     if(uploadButtonContainer.length) {
 45         /* Create UploadLinkLargeIconButton object */
 46         this.uploadButton = new UploadLinkLargeIconButton({
 47             container: uploadButtonContainer, 
 48         });        
 49     }
 50     
 51     /* Get dashboard button */
 52     var dashboardButtonContainer = contents.find('#dashboard_button');
 53     /* If this page has a dashboard button */
 54     if(dashboardButtonContainer.length) {
 55         /* Create dashboard button */
 56         this.dashboardButton = new DashboardLinkLargeIconButton({
 57             container: dashboardButtonContainer,
 58         });        
 59     }
 60     
 61     /* get settings button */
 62     var settingsButtonContainer = contents.find('#settings_button');
 63     /* if this page has a settings button */
 64     if(settingsButtonContainer.length) {
 65         /* Create settings Button */
 66         this.settingsButton = new SettingsLinkLargeIconButton({
 67             container: settingsButtonContainer, 
 68         });
 69     }
 70     
 71     
 72     //this.retrieveAndUpdateCollectionSelector();
 73     
 74     /** INitialize behavior for collection selector */
 75     collectionSelector.bind('change', function(e) {
 76         var collection_id = $(this).val();
 77         
 78         window.location = '/organize/collection/'+collection_id;
 79     });
 80 
 81     /* The Backbone view object for the collection selector */
 82     var userCollectionSelectView = new UserCollectionSelectView({
 83         id: 'collection_selector', 
 84         collection: this.page.collections,
 85         collectionSelectorOptionsTemplate: $('#collection_selector_options_template'), 
 86         collectionSelector: collectionSelector, 
 87     });
 88     
 89     userCollectionSelectView.render();
 90     this.userCollectionSelectView = userCollectionSelectView;
 91 }
