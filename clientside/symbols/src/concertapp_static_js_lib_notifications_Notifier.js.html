  7 /**
  8  *  This is an object which can notify the user in a few different ways.  Right now
  9  *  the only options that are available are modal dialogs and modal alerts.  In 
 10  *  the future, a "growl" like interface can be triggered from here.  This should
 11  *  be a singleton object, i.e. only one should be instantiated.
 12  *	@class
 13  **/
 14 function Notifier(params) {
 15     if(params) {
 16         this.init(params);
 17     }
 18 }
 19 
 20 Notifier.prototype.init = function(params) {
 21     
 22     this.modalAlertObject = new ConfirmModalAlert(params);
 23 }
 24 
 25 /**
 26  *  Use this if you just want to display words to user, and not worry about getting
 27  *  anything back.
 28  *
 29  *  @param  params        Object            {
 30  *                                              title: The title of the alert
 31  *                                              content: the content of the alert
 32  *                                              url: instead of content, the url
 33  *                                          }
 34  **/
 35 Notifier.prototype.alert = function(params) {
 36     var contentOrUrl = this.validateParams(params);
 37     if(contentOrUrl == 'content') {
 38         this.modalAlertObject.displayContent(params);
 39     }
 40     else {
 41         this.modalAlertObject.loadAndDisplayContent(params);
 42     }
 43 }
 44 
 45 /**
 46  *  Use this if you want to do something different if the user chooses yes or no.
 47  *
 48  *  @param  params        Object            {
 49  *                                              title: The title of the alert
 50  *                                              content: the content of the alert
 51  *                                              url: instead of content, the url
 52  *                                              confirmCallback: the function to 
 53  *                                                  execute when user confirms.
 54  *                                              cancelCallback: the function to 
 55  *                                                  execute when user cancels.
 56  *                                          }
 57  **/
 58 Notifier.prototype.confirm = function(params) {
 59     var contentOrUrl = this.validateParams(params);
 60     
 61     if(contentOrUrl == 'content') {
 62         this.modalAlertObject.displayContentWithConfirm(params);
 63     }
 64     else {
 65         this.modalAlertObject.loadAndDisplayContentWithConfirm(params);
 66     }
 67 }
 68 
 69 /**
 70  *  Since we need to do this for a confirm or alert.  Just makes sure user sent
 71  *  in either content or URL parameters.
 72  *
 73  *  @param  params        Object 
 74  **/
 75 Notifier.prototype.validateParams = function(params) {
 76     
 77     
 78     /* If there is no cancelCallback, use empty function */
 79     if(typeof(params.cancelCallback) == 'undefined') {
 80         params.cancelCallback = function() {
 81             
 82         };
 83     };
 84     
 85     
 86     /* Check if the user passed in content */
 87     var content = params.content;
 88     if(typeof(content) == 'undefined') {
 89         /* Check if the user passed in a url */
 90         var url = params.url;
 91         if(typeof(url) == 'undefined') {
 92             throw new Error('Error: params.url or params.content must be defined.');
 93         }
 94         else {
 95             /* URL was passed in */
 96             return 'url';
 97         }
 98     }
 99     else {
100         /* Content was passed in */
101         return 'content';
102     }
103     
104 }
