  7 /**
  8  *  The panel that contains the upload form for the user to upload multiple files.
  9  *	@class
 10  **/
 11 function UploadPanel(params) {
 12     if(params) {
 13         this.init(params);
 14     }
 15 }
 16 UploadPanel.prototype = new Panel();
 17 
 18 /**
 19  *  @param  uploadFormElement       jQuery HTMLFormElement - the actual form that
 20  *                                      will be used to send to the server.
 21  *  @param  fileChooserContainer    jQuery HTMLDivElement - the container for
 22  *                                      the file chooser <input> element.  
 23  *  @param  fileChooserTemplate     jQuery tmpl script object for creating the
 24  *                                      file chooser <input> element.
 25  *  @param  uploadFileWidgetTemplate         jQUery tmpl script object for a row in the table.
 26  *  @param  filesTable              jQUery HTMLDivElement where the files are listed.  
 27  *  @param  uploadButtonElement     jQuery HTMLButtonElement that the user
 28  *                                      will press to start uploading the files.
 29  *  @param  fileChooserLabel        jQuery HTMLLabelElement that says "select files"  
 30  *                                      or something like that.
 31  *  @param  statusContainer         jQuery HTMLDivElement that will contain the
 32  *                                      'uploading...' notification.
 33  **/
 34 UploadPanel.prototype.init = function(params) {
 35     Panel.prototype.init.call(this, params);
 36 
 37     /* We will keep the filewidgets here, indexed by their 'count' */
 38     this.files = [];
 39     
 40     /* We will also keep the filewidgets here, but remove them when they start
 41         uploading */
 42     this.filesToUpload = [];
 43     
 44     /* The form we will change each time we upload a file */
 45     var uploadFormElement = params.uploadFormElement;
 46     if(typeof(uploadFormElement) == 'undefined') {
 47         throw new Error('params.uploadFormElement is undefined');
 48     }
 49     else if(uploadFormElement.length == 0) {
 50         throw new Error('uploadFormElement not found');
 51     }
 52     this.uploadFormElement = uploadFormElement;
 53     
 54     var fileChooserContainer = params.fileChooserContainer;
 55     if(typeof(fileChooserContainer) == 'undefined') {
 56         throw new Error('params.fileChooserContainer is undefined');
 57     }
 58     else if(fileChooserContainer.length == 0) {
 59         throw new Error('fileChooserContainer not found');
 60     }
 61     this.fileChooserContainer = fileChooserContainer;
 62     
 63     var fileChooserTemplate = params.fileChooserTemplate;
 64     if(typeof(fileChooserTemplate) == 'undefined') {
 65         throw new Error('params.fileChooserTemplate is undefined');
 66     }
 67     else if(fileChooserTemplate.length == 0) {
 68         throw new Error('fileChooserTemplate not found');
 69     }
 70     this.fileChooserTemplate = fileChooserTemplate;
 71     
 72     var uploadFileWidgetTemplate = params.uploadFileWidgetTemplate;
 73     if(typeof(uploadFileWidgetTemplate) == 'undefined') {
 74         throw new Error('params.uploadFileWidgetTemplate is undefined');
 75     }
 76     else if(uploadFileWidgetTemplate.length == 0) {
 77         throw new Error('uploadFileWidgetTemplate not found');
 78     }
 79     this.uploadFileWidgetTemplate = uploadFileWidgetTemplate;
 80     
 81     var filesTable = params.filesTable;
 82     if(typeof(filesTable) == 'undefined') {
 83         throw new Error('params.filesTable is undefined');
 84     }
 85     else if(filesTable.length == 0) {
 86         throw new Error('filesTable not found');
 87     }
 88     this.filesTable = filesTable;
 89     
 90     var uploadButtonElement = params.uploadButtonElement;
 91     if(typeof(uploadButtonElement) == 'undefined') {
 92         throw new Error('params.uploadButtonElement is undefined');
 93     }
 94     else if(uploadButtonElement.length == 0) {
 95         throw new Error('uploadButtonElement not found');
 96     }
 97     this.uploadButtonElement = uploadButtonElement;
 98 
 99     var fileChooserLabel = params.fileChooserLabel;
100     if(typeof(fileChooserLabel) == 'undefined') {
101         throw new Error('params.fileChooserLabel is undefined');
102     }
103     else if(fileChooserLabel.length == 0) {
104         throw new Error('fileChooserLabel not found');
105     }
106     this.fileChooserLabel = fileChooserLabel;
107     
108     var statusContainer = params.statusContainer;
109     if(typeof(statusContainer) == 'undefined') {
110         throw new Error('params.statusContainer is undefined');
111     }
112     else if(statusContainer.length == 0) {
113         throw new Error('statusContainer not found');
114     }
115     this.statusContainer = statusContainer;
116     
117     /* This will animate the dots on the status container */
118     this.dotsAnimator = new AnimatedDots({
119         container: params.statusDots
120     });
121     
122     /* We don't need to show status container right now */
123     statusContainer.hide();
124 
125     
126     /* When upload button is pressed, start uploading */
127     uploadButtonElement.click(function(me) {
128         return function() {
129             me.startUpload();
130         };
131     }(this));
132     
133     
134     /* The current file chooser that the user can see */
135     this.currentFileChooser = null;
136     
137     /* Keep track of how many files have been chosen.  This is incremented
138         when a file is chosen. */
139     this.fileIndex = -1;
140     
141     
142     this.createNewFileChooser();
143     
144     
145 }

152 /**
153  *  Increment this.count member variable, and create the new "file chooser".  Then
154  *  put this in the appropriate location in the DOM.
155  **/
156 UploadPanel.prototype.createNewFileChooser = function() {
157     /* Increment the file index because we may add a new file */
158     var newFileIndex = ++this.fileIndex;
159     
160     /* Create new file chooser */
161     var chooser = this.fileChooserTemplate.tmpl({
162         id: newFileIndex, 
163     });
164     
165     /* When a file is chosen with this new element */
166     chooser.bind('change', function(me){
167         return function() {
168             me.addFile($(this));
169         };
170     }(this));
171     
172     /* Add chooser to DOM */
173     this.fileChooserContainer.html(chooser);    
174     
175 };

178 /**
179  *  This should be called when the user selects a file.  It will add the file to 
180  *  the queue, as well as to the 'file_list' in the DOM.  This will be done by
181  *  creating a new UploadingFileWidget, and moving the inputElement into the
182  *  widget.
183  *
184  *  @param  fileChooser        jQuery HTMLInputElement - the <input> element that 
185  *                              the user used to select this file.
186  **/
187 UploadPanel.prototype.addFile = function(fileChooser) {
188     /* The index for the current file */
189     var index = this.fileIndex;
190     
191     /* Create new UploadFileWidget */
192     var widget = new UploadFileWidget({
193         template: this.uploadFileWidgetTemplate, 
194         context: {
195             filename: fileChooser.attr('files')[0].fileName, 
196             id: index
197         },
198         /* Remove fileChooser from DOM, and send to UploadFileWidget */
199         fileChooser: fileChooser.detach(), 
200         panel: this
201     });
202 
203     
204     /* Add widget to DOM */
205     this.filesTable.append(widget.container);
206     /* Add widget to data structures */
207     this.files[index] = widget;
208     this.filesToUpload[index] = widget;
209     
210     /* Re-populate document with new file chooser (do this when we can support
211         multiple file uploads) */
212     //this.createNewFileChooser();
213     
214     /* Remove "select a file to upload" button */
215     this.fileChooserLabel.hide();
216 };
217 
218 /**
219  *  Should be called when upload button is pressed.  This will initiate the uploading
220  *  of all the widgets in the queue.
221  **/
222 UploadPanel.prototype.startUpload = function() {
223     
224     /* Hide upload button */
225     this.uploadButtonElement.hide();
226     
227     /* Hide file chooser and file chooser label form */
228     this.fileChooserContainer.hide();
229     this.fileChooserLabel.hide();
230     
231     /* Show "uploading..." or something of that nature at the top of the file list */
232     this.statusContainer.show();
233     
234     /* Start dot animation */
235     this.dotsAnimator.start();
236     
237     /* Upload first file */
238     this.uploadNext();
239 };
240 
241 /**
242  *  This should be called after startUpload has been called once, and after
243  *  each file is done uploading.
244  **/
245 UploadPanel.prototype.uploadNext = function() {
246     var next = this.filesToUpload.shift();
247     
248     next.startUpload();
249 };
