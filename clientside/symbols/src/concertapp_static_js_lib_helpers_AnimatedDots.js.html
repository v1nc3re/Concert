  7 /**
  8  *  This is a class that can be used when we want to animate some dots
  9  *  .
 10  *  ..
 11  *  ...
 12  *  ^ Like that.
 13  *  Sometimes I like to do this because it is much cheaper than a graphical
 14  *  loading notification.
 15  *	@class
 16  **/
 17 function AnimatedDots(params) {
 18     if(params) {
 19         this.init(params);
 20     }
 21 }
 22 
 23 /**
 24  *  @param  container       jQuery HTMLDivElement/HTMLSpanElement...Anything really.
 25  *  @param  numDots         Number - defaults to 3  
 26  *  @param  animationSpeed        Number - defaults to 500  
 27  **/
 28 AnimatedDots.prototype.init = function(params) {
 29 
 30     var container = params.container;
 31     if(typeof(container) == 'undefined') {
 32         throw new Error('params.container is undefined');
 33     }
 34     else if(container.length == 0) {
 35         throw new Error('container not found');
 36     }
 37     this.container = container;
 38     
 39     var numDots = params.numDots;
 40     if(typeof(numDots) == 'undefined') {
 41         numDots = 3;
 42     }
 43     else if(numDots < 1) {
 44         throw new Error('params.numDots must be at least 1');
 45     }
 46     this.numDots = numDots;
 47     
 48     var animationSpeed = params.animationSpeed;
 49     if(typeof(animationSpeed) == 'undefined') {
 50         animationSpeed = 500;
 51     }
 52     this.animationSpeed = animationSpeed;
 53 
 54     /* We will keep track of the contents in a variable so we don't have to both
 55         check the DOM, and update the DOM. */
 56     this.contents = container.html();
 57     
 58     /* This is what the final step of the animation will look like, it will just be
 59         numDots amount of period characters */
 60     var finalContents = '';
 61     for(i = 0; i < numDots; i++) {
 62         finalContents += '.';
 63     }
 64     this.finalContents = finalContents;
 65     
 66     /* Keep the interval so we can stop it */
 67     this.interval = null;
 68     
 69 }
 70 
 71 /**
 72  *  This function is called every this.animationSpeed ms, from within the interval
 73  *  it animates the dots one more step further.
 74  **/
 75 AnimatedDots.prototype.animate = function() {
 76     var currentContents = this.contents;
 77     var finalContents = this.finalContents;
 78     
 79     var newContents = null;
 80     
 81     if(currentContents == finalContents) {
 82         newContents = '';
 83     }
 84     else {
 85         newContents = currentContents + '.';
 86     }
 87     
 88     this.contents = newContents;
 89     this.container.html(newContents);
 90 }
 91 
 92 /**
 93  *  This will start the animation, and should be called from outside.
 94  **/
 95 AnimatedDots.prototype.start = function() {
 96     this.interval = setInterval(function(me) {
 97         return function() {
 98             me.animate();
 99         };
100     }(this), this.animationSpeed);
101 }
102 
103 /**
104  *  This will stop the animation, and should be called from outside.
105  **/
106 AnimatedDots.prototype.stop = function() {
107     clearInterval(this.interval);
108     this.interval = null;
109 }
