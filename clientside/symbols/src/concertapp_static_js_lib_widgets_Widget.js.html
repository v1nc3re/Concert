  7 /**
  8  *  This contains general stuff that needs to take place for any widget on the UI.
  9  *	@class
 10  **/
 11 function Widget(params) {
 12     if(params) {
 13         this.init(params);
 14     }
 15 }
 16 
 17 /**
 18  *  Create the DOM elements associated with this widget using a template.  The
 19  *  container member variable is then set automatically for the widget.  The widget
 20  *  is not inserted into the DOM.
 21  *
 22  *  @param  params.template        jQuery tmpl object   -   THe template.
 23  *  @param  params.context        Object     - the context to send to template.
 24  *  @param  params.panel        Panel object that we belong to.  
 25  **/
 26 Widget.prototype.init = function(params) {
 27     var template = params.template;
 28     if(typeof(template) == 'undefined') {
 29         throw new Error('params.template is undefined');
 30     }
 31     else if(template.length == 0) {
 32         throw new Error('template not found');
 33     }
 34     
 35     var context = params.context;
 36     if(typeof(context) == 'undefined') {
 37         throw new Error('params.context is undefined');
 38     }
 39     this.context = context;
 40     
 41     var panel = params.panel;
 42     if(typeof(panel) == 'undefined') {
 43         throw new Error('params.panel is undefined');
 44     }
 45     this.panel = panel;
 46     
 47     this.id = context.id; 
 48     
 49     this.container = template.tmpl(context);
 50 }
