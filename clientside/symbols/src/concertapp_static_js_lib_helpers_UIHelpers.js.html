  8 /**
  9  *  This will auto clear the fields that have a class "autoClear" on them.
 10  *  The placeholder text is from the "data-placeholder" attribute of the 
 11  *  element.
 12  **/
 13 function initializeAutoClearFieldBehavior(){
 14     
 15     /* Make jQuery local for performance */
 16     var $ = jQuery;
 17     
 18     $('input.autoClear').live('focus', function(event) {
 19         /*  If the placeholder attribute hasn't been set yet */
 20         if (!$(this).data('placeholder')) {
 21             /* save what is in the field */
 22             $(this).data('placeholder', $(this).val());
 23         }
 24         
 25         /* If the text field was just focused and has the placeholder text */
 26         if ($(this).val() == $(this).data('placeholder')) {
 27             /* remove placeholder text */
 28             $(this).val('');
 29         }
 30     }).live('blur', function(event) {        
 31         /* When field is blurred, put placeholder text back */
 32         if ($.trim($(this).val()) == '') {
 33             $(this).val($(this).data('placeholder'));
 34         }
 35     });
 36     
 37 }

 39 /**
 40  *  Preloads all images via an image beacon to the server.
 41  *
 42  *  @param  urls        Array 
 43  **/
 44 function preloadImages(urls) {
 45     var images = [];
 46     for(i = 0, il = urls.length; i < il; i++) {
 47         images.push($('<img />').attr('src', urls[i]));
 48     }
 49 }

 51 /**
 52  *  Detect browser compatibility problems, and notify user of any issues.  Should
 53  *  be called on intro page, and the user should not be allowed to go further if
 54  *  there are problems.
 55  **/
 56 function detectBrowserCompatibility() {
 57     var notifier = com.concertsoundorganizer.notifier;
 58     
 59     /**
 60      *  Detect audio compatibility, and throw notifications to the user if necessary.
 61      **/
 62     if(Modernizr.audio) {
 63         if(Modernizr.audio.ogg) {
 64             com.concertsoundorganizer.compatibility.audiotype = 'ogg';
 65         }
 66         else if(Modernizr.audio.mp3) {
 67             com.concertsoundorganizer.compatibility.audiotype = 'mp3';
 68         }
 69         else {
 70             /* TODO: Handle this case */
 71             notifier.alert({content: 'Site will not work.  75         /* TODO: Handle old browsers here */

 80 /**
 81  *  Takes an array of DOM nodes, and a container.  It will replace all of the nodes
 82  *  in the container with the given.
 83  *
 84  *  @param  nodes        Array  -   of DOM nodes to add to container
 85  *  @param  container        jQuery DOM element  
 86  **/
 87 function domElementsReplace(nodes, container) {
 88     /* Empty container */
 89     container.empty();
 90     
 91     /* Loop through all of the nodes, and add them to a fragment */
 92     var frag = document.createDocumentFragment();
 93     for(i = 0, il = nodes.length; i < il; i++) {
 94         frag.appendChild(nodes[i]);
 95     }
 96     container.append(frag);    
 97 }
