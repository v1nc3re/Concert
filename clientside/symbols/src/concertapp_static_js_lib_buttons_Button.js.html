  7 /**
  8  *  All buttons will inherit from this class.  Contains functionality that is
  9  *  necessary for all buttons.
 10  *	@class
 11  **/
 12 function Button(params) {
 13     if(params) {
 14         this.init(params);
 15     }
 16 }
 17 
 18 
 19 /**
 20  *  Initializes behavior for buttons.
 21  *
 22  *  @param  params.container            jQuery HTMLDivElement - The DOM element that
 23  *                                      all of the buttons behaviors will be assigned
 24  *                                      to.
 25  *
 26  **/
 27 Button.prototype.init = function(params) {
 28 
 29     /* Container where all of the behavior is assigned to is required */
 30     var container = params.container;
 31     if(typeof(container) == 'undefined') {
 32         throw new Error('params.container is undefined');
 33     }
 34     else if(container.length == 0) {
 35         throw new Error('container not found for '+container.selector);
 36     }
 37     this.container = container;
 38 
 39     
 40 
 41 
 42     /* Assign behavior to container */
 43     container.mouseout(function(me){
 44         return function() {
 45             me.hoverOut();
 46         }
 47     }(this)); 
 48     container.mouseover(function(me){
 49         return function() {
 50             me.hoverIn();
 51         }
 52     }(this));
 53     container.click(function(me){
 54         return function(){
 55             me.click();
 56         };
 57     }(this));
 58 
 59     
 60     
 61 }
