  7 /**
  8  *  The widget that represents a file in the list of files the user has selected for
  9  *  uploading.
 10  *	@class
 11  **/
 12 function UploadFileWidget(params) {
 13     if(params) {
 14         this.init(params);
 15     }
 16 }
 17 UploadFileWidget.prototype = new Widget();
 18 
 19 /**
 20  *  @param  fileChooser         jQuery HTMLInputElement - that was used to select
 21  *                                  this file.
 22  **/
 23 UploadFileWidget.prototype.init = function(params) {
 24     Widget.prototype.init.call(this, params);
 25 
 26     var fileChooser = params.fileChooser;
 27     if(typeof(fileChooser) == 'undefined') {
 28         throw new Error('params.fileChooser is undefined');
 29     }
 30     else if(fileChooser.length == 0) {
 31         throw new Error('fileChooser not found');
 32     }
 33     this.fileChooser = fileChooser;
 34     
 35     
 36     
 37     var container = this.container;
 38     var id = this.id;
 39     
 40     /* Get the file loading progress element */
 41     this.progressElement = container.find('#uploadfilewidget_progress-'+id);
 42     
 43     /* The collection select element */
 44     this.collectionSelectElement = container.find('#uploadfilewidget_collection_selector-'+id);
 45     
 46     
 47     /* We will save the unique upload_id for our file */
 48     this.upload_id = null;
 49     
 50     /* Are we currently tracking our progress? */
 51     this.progressTracking = false;
 52     
 53     /* The delay between progress tracking requests */
 54     this.progressTrackingDelay = 400;
 55     
 56 }
 57 
 58 /**
 59  *  What we should do in the event of an upload error.
 60  **/
 61 UploadFileWidget.prototype.handleUploadError = function(data) {
 62     com.concertsoundorganizer.notifier.alert({
 63         'title': 'Error', 
 64         'content': 'An error occurred while uploading.  Sorry!\n\n'+data
 65     });
 66     
 67     
 68     this.stopProgressTracking();
 69     
 70     this.panel.handleUploadError();
 71 };

 74 /**
 75  *  This should be called when the file associated with this widget is to be
 76  *  uploaded.
 77  **/
 78 UploadFileWidget.prototype.startUpload = function() {
 79     /* Reserve upload_id, then start uploading file */
 80     this.reserveUploadId();
 81 } 
 82 
 83 /**
 84  *  Reserves an upload_id in the server's cache, so we can keep track of 
 85  *  the progress of our upload.  Once this is complete, we move on to actually
 86  *  uploading the file.
 87  **/
 88 UploadFileWidget.prototype.reserveUploadId = function() {
 89     /* Get actual upload form that we must inject our fileChooser <input> element
 90         into */
 91     var actualUploadForm = this.panel.uploadFormElement;
 92     
 93     /* Put our file chooser into this form */
 94     actualUploadForm.html(this.fileChooser);
 95     
 96     /* Get unique upload id */
 97     $.ajax({
 98         dataType: 'json',
 99         url: 'get_id/', 
100         success: function(me) {
101             return function(data, status, xhr) {
102                 if(typeof(data.status) != 'undefined' && data.status == 'success') {
103                     /* We have a unique upload_id now. */
104                     me.upload_id = data.upload_id;
105                 
106                     /* Move forward */
107                     me.uploadFile();
108                 }
109                 else {
110                     me.handleUploadError(data);
111                 }
112             };
113         }(this)
114     });
115 };
116 
117 /**
118  *  Actually upload the file.  This can only occur after we have received an
119  *  upload_id from the server.
120  **/
121 UploadFileWidget.prototype.uploadFile = function() {
122     
123     /* Make sure we have an upload_id */
124     if(this.upload_id == null) {
125         throw new Error('this.upload_id has not been set.  Please reserve an upload_id before uploading the file.');
126     }
127     
128     /* Get actual upload form that we must inject our fileChooser <input> element
129         into */
130     var actualUploadForm = this.panel.uploadFormElement;
131     
132     /* Put our file chooser into this form */
133     actualUploadForm.html(this.fileChooser);
134     
135     /* Put our upload_id into this form as well*/
136     actualUploadForm.append($('<input />').attr({
137         type: 'hidden', 
138         name: 'upload_id', 
139         id: 'upload_id',         
140         value: this.upload_id, 
141     }));
142     
143     /* Put our collection select element into the form */
144     actualUploadForm.append(this.collectionSelectElement);
145     
146     /* Get unique upload id */
147     actualUploadForm.ajaxSubmit({
148         /* This will return an XML document (since we are psuedo-hacking a form) */
149         dataType: 'text',
150         url: '/audio/upload/?upload_id='+this.upload_id,
151         beforeSubmit: function(arr, form, options) {
152         },
153         success: function(me) {
154             return function(data, status, xhr) {
155                 if(data.match('<pre style="word-wrap: break-word; white-space: pre-wrap;">success</pre>')) {
156                     
157                     /* The file was hopefully uploaded */
158                     me.encodingWaiting = false;
159                     
160                     com.concertsoundorganizer.notifier.alert({
161                         title: 'Success!', 
162                         content: 'Your file uploaded successfully.', 
163                     });
164                     
165                 }
166                 else {
167                     me.handleUploadError(data);
168                 }
169             };
170         }(this)
171     });
172     
173     this.startProgressTracking();
174     
175 };

178 /**
179  *  This should be called initially when the file is starting to upload.  It will
180  *  start the checking of upload progress.
181  **/
182 UploadFileWidget.prototype.startProgressTracking = function() {
183     this.progressTracking = true;
184     
185     this.checkOnProgress();
186 };
187 
188 /**
189  *  send a request to the server with our upload_id to receive file upload progress
190  *  back.
191  **/
192 UploadFileWidget.prototype.checkOnProgress = function() {
193     /* We will send a request to the server with our upload_id */
194     $.ajax({
195         url: 'progress/'+this.upload_id,
196         dataType: 'json',
197         success: function(me) {
198             return function(data, status, xhr) {
199                 me.updateProgress(data);
200 
201                 /* In progressTrackingDelay ms, check on progress again */
202                 if(me.progressTracking) {
203                     setTimeout(function(me){
204                         return function() {
205                             me.checkOnProgress();
206                         };
207                     }(me), me.progressTrackingDelay);        
208                 }
209                 else if(me.encodingWaiting) {
210                     me.encodingWaitingNotification();
211                 }
212             };
213         }(this)
214     });
215 };

218 /**
219  *  Should be called whenever the progress slider is to be updated.
220  **/
221 UploadFileWidget.prototype.updateProgress = function(data) {
222     
223     var progress = null;
224     if(data == null) {
225         /* Complete! */
226         progress = 1;
227         /* Stop progress tracking */
228         this.progressTracking = false;
229         /* We are waiting for encoding to finish */
230         this.encodingWaiting = true;
231     }
232     else {
233         var uploaded = data.uploaded;
234         var length = data.length;
235         
236         if(uploaded == 0 && length == 0) {
237             progress = 0;
238         }
239         else {
240             progress = uploaded/length;
241         }
242         
243     }
244     
245     /* Update progress slider */
246     this.progressElement.attr('value', progress);
247 };
248 
249 /**
250  *  This should be called when the file has finished uploading, and we are
251  *  just waiting for it to encode.  This displays the "inifinite" progress bar.
252  **/
253 UploadFileWidget.prototype.encodingWaitingNotification = function() {
254     this.progressElement.removeAttr('value');
255     
256 };

262 /**
263  *  This should be called when the upload is complete.
264  **/
265 UploadFileWidget.prototype.stopProgressTracking = function() {
266     this.progressTracking = false;
267 };
