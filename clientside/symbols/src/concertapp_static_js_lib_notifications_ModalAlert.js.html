  7 /**
  8  *  A ModalAlert object is a "window" that pops up to alert the user.  Any content
  9  *  can be injected into this window.  The intent is for only one of these objects
 10  *  to be instantiated ever, but I suppose you could have multiple ModalAlert objects
 11  *  if you wanted to.
 12  *
 13  *	@class
 14  **/
 15 function ModalAlert(params) {
 16     if(params) {
 17         this.init(params);
 18     }
 19 }
 20 
 21 ModalAlert.prototype.init = function(params) {
 22     /**
 23      *  Initialize behavior for closing modal window
 24      **/
 25     var closeButton = $('button#modal-overlay-close-button');
 26     closeButton.live('click',function(me) {
 27         return function() {
 28             $(me).trigger('cancel');
 29             me.close();
 30         }
 31     }(this));
 32     this.closeButton = closeButton;
 33     
 34     $(document).keyup(function(me){
 35         return function(event){
 36             if(me.isOpen) {
 37                 /* If escape key was pressed */
 38                 if(event.keyCode == 27) {
 39                     $(me).trigger('cancel');
 40                     me.close();
 41                 }
 42             }
 43         };
 44     }(this));
 45     
 46     /*  Is modal window open? it is initially closed  */
 47     var open = false; 
 48     this.open = open;
 49     
 50     /* For right now, we can only use our HTML, but these should really be
 51         passed in to make this code flexible */
 52     var pane = $('div#modal-overlay-pane');
 53     this.pane = pane;
 54     
 55     var paneHeader = $('h2#modal-overlay-header');
 56     this.paneHeader = paneHeader;
 57     
 58     var paneContent = $('div#modal-overlay-content');
 59     this.paneContent = paneContent;
 60     
 61     var container = $('div#modal-overlay');
 62     this.container = container;
 63     
 64 }

 67 /**
 68  *  Display a modal window with the content sent into the content argument.
 69  *  position in the middle of the screen.
 70  *
 71  *  @param  String/HTMLDivElement       params.content: The content for the window
 72  *  @param  String                      params.title: The content for the title, 
 73  *                                          defaults to "Alert".
 74  *  @param  String                      params.url: The URL of the content to display  
 75  *  @param  Boolean                     params.noOptions: if true, no options will be
 76  *                                          displayed.
 77  **/
 78 ModalAlert.prototype.displayContent = function(params) {
 79     var $ = jQuery;
 80     
 81     var overlayPane = this.pane;
 82     overlayPane.hide();
 83     
 84     var paneHeader = this.paneHeader;
 85     var paneContent = this.paneContent;
 86     
 87     var content = params.content;
 88     if(typeof(content) == 'string') {
 89         paneContent.html(content);
 90     }
 91     else if(typeof(content) == 'HTMLDivElement') {
 92         paneContent.append(content);
 93     }
 94     else if(typeof(content) == 'undefined') {
 95         var url = params.url;
 96         if(typeof(url) == 'undefined') {
 97             throw new Error('Error: params.content or params.url must be defined');
 98         }
 99         else {
100             this.loadAndDisplayContent({
101                 'url': url,
102                 'title': title
103             });
104             return;
105         }
106     }
107     else {
108         throw new Error('content must be a string or HTMLDivElement');
109     }
110     
111     var title = params.title;
112     if(typeof(title) == 'undefined') {
113         title = 'Alert';
114     }
115     paneHeader.html(title);
116     
117     
118     
119     this.container.show();
120     
121     
122     /* Should we hide all options? */
123     var noOptions = null;
124     if(typeof(params.noOptions) == 'undefined') {
125         noOptions = false;
126     }
127     else {
128         noOptions = true;
129     }
130     
131     if(noOptions) {
132         /* Hide close button */
133         this.closeButton.hide();
134     }
135     else {
136         /* Make sure closeButton is shown */
137         this.closeButton.show();
138     }
139     
140     
141     /* position pane in center of window */
142     var $window = $(window);
143     var windowWidth = $window.width();
144     var windowHeight = $window.height();
145     var paneWidth = overlayPane.width();
146     var paneHeight = overlayPane.height();
147     
148     
149     overlayPane.css({
150         'top': (windowHeight/2)-(paneHeight/2),
151         'left': (windowWidth/2)-(paneWidth/2)
152     });
153     
154     /* Animate entry */
155     overlayPane.fadeIn('slow');
156     
157     this.isOpen = true;
158 }

161 /**
162  *  Load a partial (html) file sent in as argument, and display that in the
163  *  modal window.
164  *
165  *  @param          Object          params {
166  *                      String          url: The URL of the content
167  *                      String          title: The title of the modal alert
168  *                                  }
169  **/
170 ModalAlert.prototype.loadAndDisplayContent = function(params) {
171     /* If there was no callback, we are just displaying the content */
172     if(typeof(params.callback) == 'undefined') {
173         var title = params.title;
174         params.callback = function(me, title) {
175             return function(data) {
176                 me.displayContent({content: data, title: title, });
177             };
178         }(this, title);        
179     }
180     
181     this.loadAndExecuteCallback(params)
182 }
183 
184 /**
185  *  Loads a partial defined by params.url, and fires callback stored
186  *  in params.callback, sending data retrieved.
187  *
188  *  @param  params        Object  {
189  *                          url: The URL of the content
190  *                          callback: The callback function that takes a data arg
191  *                          }
192  **/
193 ModalAlert.prototype.loadAndExecuteCallback = function(params) {
194     /* Show loading notification */
195     this.loading(true);
196     
197     $.ajax({
198         'url': params.url,
199         'success': function(me, callback) {
200             return function(data, textStatus) {
201                 if(textStatus == 'success') {
202                     me.loading(false);
203                     callback(data)
204                 }
205                 else {
206                     throw new Error('An error has occurred');
207                 }
208             };
209         }(this, params.callback),
210         'error': function(XMLHttpRequest, textStatus, errorThrown) {
211             throw new Error('Request failed.  textStatus: '+textStatus+', error: '+errorThrown);
212         }
213     });   
214     
215 }

217 /**
218  *  Method should be called when modal window is to close.
219  **/
220 ModalAlert.prototype.close = function() {
221     this.container.hide();
222     
223     this.isOpen = false;
224 }
225 
226 /**
227  *  Should be called when a loading window is to appear.  The argument, show, means
228  *  that if set to true, the loading box will be shown.  If set to false, the 
229  *  loading animation will stop and window will be closed.
230  *
231  *  @param  show        Boolean     true: show loading, false: hide
232  **/
233 ModalAlert.prototype.loading = function(show) {
234     if(show) {
235         /* Show modal with loading notification */
236         this.displayContent({'title': 'Loading', 'content': '', 'noOptions': true});
237         
238         this.loadingAnimationInterval = setInterval(function(me) {
239             return function() {
240                 me.animateLoadingNotification();
241             };
242         }(this), 500);
243         
244     }
245     else {
246         /* Stop loading notification animation */
247         clearInterval(this.loadingAnimationInterval);
248         this.close();
249     }
250 }

252 /**
253  *  Once a loading notification has been set, this function runs every 500ms.
254  *  It just animates the periods after the word "Loading" in the title of the
255  *  modal box.
256  **/
257 ModalAlert.prototype.animateLoadingNotification = function() {
258     var title = this.paneHeader;
259     
260     var transitions = {
261         'Loading': 'Loading.', 
262         'Loading.': 'Loading..', 
263         'Loading..': 'Loading...',
264         'Loading...': 'Loading',
265     }
266     
267     title.html(transitions[title.html()]);
268 }
