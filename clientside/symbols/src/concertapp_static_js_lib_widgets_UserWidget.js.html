  7 /**
  8  *  A user widget is typically how a user is displayed on the UI.
  9  *  @class
 10  **/
 11 function UserWidget(params) {
 12     if(params) {
 13         this.init(params);
 14     }
 15 }
 16 //UserWidget.prototype = new Widget();
 17 
 18 UserWidget.prototype.init = function(params) {
 19     //Widget.prototype.init.call(this, params);
 20     
 21     var container = params.container;
 22     if(typeof(container) == 'undefined') {
 23         throw new Error('params.container is undefined');
 24     }
 25     else if(container.length == 0) {
 26         throw new Error('container not found');
 27     }
 28     this.container = container;
 29 
 30     
 31 
 32     var usernameContainer = container.children('.user_widget_username');
 33     if(typeof(usernameContainer) == 'undefined') {
 34         throw new Error('usernameContainer is undefined');
 35     }
 36     else if(usernameContainer.length == 0) {
 37         throw new Error('malformed HTML: usernameContainer not found');
 38     }
 39     this.usernameContainer = usernameContainer;
 40 
 41     
 42     
 43 }
