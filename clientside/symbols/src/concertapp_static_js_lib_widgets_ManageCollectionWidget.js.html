  7 /**
  8  *  A collection widget is a widget that is displayed in a collections list.  It
  9  *  has information about the collection, as well as options to delete or leave
 10  *  the collection.
 11  *	@class
 12  **/
 13 function ManageCollectionWidget(params) {
 14     if(params) {
 15         this.init(params);
 16     }
 17 }
 18 ManageCollectionWidget.prototype = new Widget();
 19 
 20 /**
 21  **/
 22 ManageCollectionWidget.prototype.init = function(params) {
 23     Widget.prototype.init.call(this, params);
 24 
 25     var container = this.container;
 26     
 27     /* Collection id is easier to understand here */
 28     var collection_id = this.id;
 29     this.collection_id = this.id;
 30     
 31     var nameElement = container.find('#user_collection_name-'+collection_id);
 32     if(nameElement.length == 0) {
 33         throw new Error('Element #user_collection_name-'+collection_id+' does not exist!');
 34     }
 35     this.nameElement = nameElement;
 36     
 37     
 38     var membersElement = container.find('#user_collection_members-'+collection_id)
 39     if(membersElement.length == 0) {
 40         throw new Error('Element #user_collection_members-'+collection_id+' does not exist!');
 41     }
 42     this.membersElement = membersElement;
 43     
 44     
 45     /* The deleteButton on the widget */
 46     var deleteButton = container.find('#delete_collection-'+collection_id);
 47     if(deleteButton.length) {
 48         this.deleteButton = deleteButton;
 49         
 50         /* Initialize delete behavior here, because this will not change 
 51             dynamically */
 52         deleteButton.click(function(me){
 53             return function() {
 54                 me.deleteCollectionConfirm();
 55             };
 56         }(this));
 57     }
 58     
 59     /* The leave button on the widget */
 60     var leaveButton = container.find('#leave_collection-'+collection_id);
 61     if(leaveButton.length) {
 62         this.leaveButton = leaveButton;
 63         
 64         /* Initialize leave behavior here, because this will not change */
 65         leaveButton.click(function(me){
 66             return function() {
 67                 me.leaveCollection();
 68             };
 69         }(this));
 70     }
 71     
 72     /* The revoke request button on the widget */
 73     var revokeButton = container.find('#revoke_request-'+collection_id);
 74     if(revokeButton.length) {
 75         this.revokeButton = revokeButton;
 76         
 77         /* When revoke button is clicked, revoke join request */
 78         revokeButton.click(function(me) {
 79             return function() {
 80                 me.revokeRequest();
 81             };
 82         }(this));
 83     }
 84     
 85     /* For all of the user collection requests, create a CollectionRequestWidget*/
 86     var collectionRequestWidgets = [];
 87     
 88     
 89     var collectionRequestContainers = container.children('.user_collection_requests').children();
 90 
 91     for(var i = 0, il = collectionRequestContainers.length; i < il; i++) {
 92         collectionRequestWidgets.push(new CollectionRequestWidget({
 93             container: $(collectionRequestContainers[i]),
 94             panel: this.panel, 
 95         }));
 96     }
 97     
 98     
 99 };
100 
101 /**
102  *  This will be called when the user is revoking his/her request to join a
103  *  collection.
104  **/
105 ManageCollectionWidget.prototype.revokeRequest = function() {
106     var collection_id = this.collection_id;
107     this.panel.toggleLoadingNotification();
108     
109     $.getJSON('revoke/'+collection_id, 
110         function(me){ 
111             return function(data, status){
112                 if(status == 'success' && data.status == 'success') {
113                     me.panel.toggleLoadingNotification();
114                     me.panel.retrieveAndUpdateCollections();
115                 }
116                 else {
117                     com.concertsoundorganizer.notifier.alert({
118                         title: "Error", 
119                         content: "An error occurred: "+data.notification, 
120                     });
121                 }
122             }; }(this));
123 };

127 /**
128  *  This should be called when user presses the delete button for a collection.
129  *  Just prompts them "Are you sure?" about deleting the collection.
130  **/
131 ManageCollectionWidget.prototype.deleteCollectionConfirm = function() {
132     /* Confirm with user */
133     com.concertsoundorganizer.notifier.confirm({
134         'title': 'Are you sure?',
135         'content': 'Are you sure you want to delete collection "'+this.nameElement.html()+'"?<br />All related content will be removed from Concert.',
136         'confirmCallback': function(me) {
137             return function() {
138                 me.deleteCollection();
139             };
140         }(this)/*,
141         'cancelCallback': function(collection_id) {
142             return function() {
143 
144             };
145         }(collection_id)*/
146     });
147 }
148 
149 /**
150  *  Should be called when collection is actually to be deleted.  This is serious 
151  *  stuff, and will error on backend if user doesn't have proper permissions.
152  **/
153 ManageCollectionWidget.prototype.deleteCollection = function() {
154     /* we are serious about deleting this collection. */
155     $.ajax({
156         url: 'delete/',
157         data: { id: this.collection_id },
158         type: 'POST',
159         success: function(me) {
160             return function(data, status, xhr) {
161                 if(data == 'success') {
162                     com.concertsoundorganizer.notifier.alert({
163                         title: 'Success!',
164                         content: 'The collection has been deleted.'
165                     });
166 
167                     /* Remove me from panel */
168                     me.panel.deleteCollection(me.collection_id);
169                 }
170                 else {
171                     com.concertsoundorganizer.notifier.alert({
172                         title: 'Error', 
173                         content: 'An error has occurred.'
174                     });
175                 }
176             };
177         }(this)
178     });
179 }
