  7 /**
  8  *  Just a large icon button that redirects to a URL when clicked.
  9  *	@class
 10  **/
 11 function LinkLargeIconButton(params) {
 12     if(params) {
 13         this.init(params);
 14     }
 15 }
 16 LinkLargeIconButton.prototype = new LargeIconButton();
 17 
 18 LinkLargeIconButton.prototype.init = function(params) {
 19     LargeIconButton.prototype.init.call(this, params);
 20 
 21     /* Get data-href attribute from container.  This will define where the page 
 22         redirects to */
 23     var href = this.container.attr('data-href');
 24     if(typeof(href) == 'undefined') {
 25         throw new Error('Element #'+container.attr('id')+'\'s data-href attribute is undefined');
 26     }
 27     this.href = href;
 28     
 29 }
 30 
 31 /**
 32  *  When button is clicked, just redirect to the location specified in the data-href
 33  *  attribute of the container.
 34  **/
 35 LinkLargeIconButton.prototype.click = function() {
 36     LargeIconButton.prototype.click.call(this);
 37     
 38     window.location = this.href;
 39 }
