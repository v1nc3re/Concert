  7 /**
  8  *  The panels and widgets for the organize page.
  9  *	@class
 10  **/
 11 function OrganizePage(params) {
 12     if(params) {
 13         this.init(params);
 14     }
 15 }
 16 OrganizePage.prototype = new LoggedInPage();
 17 
 18 OrganizePage.prototype.init = function(params) {
 19     LoggedInPage.prototype.init.call(this, params);
 20 
 21     /*  Create waveform viewer panel */
 22     this.viewerPanel = new WaveformViewerPanel({
 23         page: this, 
 24         container: $('#waveform_viewer_panel'), 
 25     });
 26     
 27     
 28     /* Create the audio list panel */    
 29     this.audioListPanel = new AudioListPanel({
 30         page: this, 
 31         container: $('#audio_list_panel'), 
 32         fileWidgetTemplate: $('#file_widget_template'), 
 33         segmentWidgetTemplate: $('#segment_widget_template')
 34     });
 35     
 36 }
