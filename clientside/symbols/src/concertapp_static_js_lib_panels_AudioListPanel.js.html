  7 /**
  8  *  Displays the segment or clip list within the audio organize page.
  9  *	@class
 10  **/
 11 function AudioListPanel(params) {
 12     if(params) {
 13         this.init(params);
 14     }
 15 }
 16 AudioListPanel.prototype = new Panel();
 17 
 18 /**
 19  *  @param  params.fileWidgetTemplate           The jquery tmpl script for a file 
 20  *                                              widget  
 21  *  @param  params.segmentWidgetTemplate        the jQuery tmpl script for a segment
 22  *                                              widget.
 23  *  @param  params.mode                         String (optional) 'files' or
 24  *                                              'segments' will be the initial mode
 25  *                                              of this panel.
 26  **/
 27 AudioListPanel.prototype.init = function(params) {
 28     Panel.prototype.init.call(this, params);
 29 
 30     var container = this.container;
 31     var header = this.header;
 32     
 33     /* Get the two different headers */
 34     var segmentsHeader = header.children('#segments_header');
 35     var filesHeader = header.children('#files_header');
 36     
 37     if(typeof(segmentsHeader) == 'undefined' || segmentsHeader.length == 0) {
 38         throw new Error('segmentsHeader not found: '+segmentsHeader.selector);
 39     }
 40     if(typeof(filesHeader) == 'undefined' || filesHeader.length == 0) {
 41         throw new Error('filesHeader not found: '+filesHeader.selector);
 42     }
 43     this.segmentsHeader = segmentsHeader;
 44     this.filesHeader = filesHeader;
 45     
 46     var fileWidgetTemplate = params.fileWidgetTemplate;
 47     if(typeof(fileWidgetTemplate) == 'undefined') {
 48         throw new Error('params.fileWidgetTemplate is undefined');
 49     }
 50     else if(fileWidgetTemplate.length == 0) {
 51         throw new Error('fileWidgetTemplate not found');
 52     }
 53     this.fileWidgetTemplate = fileWidgetTemplate;
 54     
 55     
 56     var segmentWidgetTemplate = params.segmentWidgetTemplate;
 57     if(typeof(segmentWidgetTemplate) == 'undefined') {
 58         throw new Error('params.segmentWidgetTemplate is undefined');
 59     }
 60     else if(segmentWidgetTemplate.length == 0) {
 61         throw new Error('segmentWidgetTemplate not found');
 62     }
 63     this.segmentWidgetTemplate = segmentWidgetTemplate;
 64     
 65     /** Audio panel mode switcher buttons **/
 66     /* These are probably going to be a different kind of button in the future */
 67     var audioSwitcherSegmentsButton = new LargeIconButton({
 68         container: header.children('#audiolist_switcher_segments_button')
 69     });
 70     var audioSwitcherFilesButton = new LargeIconButton({
 71         container: header.children('#audiolist_switcher_files_button')
 72     });
 73     
 74     this.audioSwitcherSegmentsButton = audioSwitcherSegmentsButton;
 75     this.audioSwitcherFilesButton = audioSwitcherFilesButton;
 76 
 77     
 78     
 79 
 80     
 81     /*  Mode is either 'files' or 'segments', as is reflected on the UI */
 82     var mode = params.mode;
 83     if(typeof(mode) == 'undefined') {
 84         mode = 'segments';
 85     }
 86     this.mode = mode;
 87     
 88     /* Initialize button behavior to switch mode */
 89     audioSwitcherFilesButton.click = function(me) {
 90         return function() {
 91             me.showFiles();            
 92         };
 93     }(this);
 94     audioSwitcherSegmentsButton.click = function(me) {
 95         return function() {
 96             me.showSegments();            
 97         };
 98     }(this);
 99 
100     
101     
102     
103     /* All of our file widgets */
104     this.fileWidgets = null;
105     /* All of our file widget DOM nodes */
106     this.fileWidgetNodes = null;
107     
108     /* Our segment widgetes */
109     this.segmentWidgets = null;
110     /* And their respective nodes */
111     this.segmentWidgetNodes = null;
112     
113     /*  The audio and segment data objects (retrieved from server) */
114     this.audioData = null;
115     this.segmentData = null;
116     
117 
118     this.retrieveData();
119 }
120 
121 /**
122  *  Get all audio data from the server for the current collection.
123  **/
124 AudioListPanel.prototype.retrieveData = function() {
125     this.toggleLoadingNotification();
126     
127     $.getJSON(
128         /* window.location here has the collection id encoded in it */
129         window.location+'/audio', 
130         function(me) {
131             return function(data, status, xhr){
132                 me.processData(data);
133                 me.toggleLoadingNotification();
134             };
135         }(this)
136     );
137 };
138 
139 /**
140  *  After we have retrieved all of the audio data from the server,
141  *  process it.  Here is where all of the widgets are created.
142  *
143  *  @param  data        Object  raw data received from server
144  **/
145 AudioListPanel.prototype.processData = function(data) {
146     var audioDataServer = data.audio_objects;
147     var segmentDataServer = data.segment_objects;
148     
149     
150     /* Stuff we will need */
151     var fileWidgetTemplate = this.fileWidgetTemplate;
152     var segmentWidgetTemplate = this.segmentWidgetTemplate;
153     
154     /* Where we'll store the audio and segment objects indexed by id */
155     var audioData = [];
156     var segmentData = [];
157     
158     /*  First we need to make our own audioData and segmentData arrays that are 
159         indexed by the object's id.
160     */
161     for(i = 0, il = audioDataServer.length; i < il; i++) {
162         var audio = audioDataServer[i];
163         
164         audioData[audio.id] = audio;
165     }
166     for(i = 0, il = segmentDataServer.length; i < il; i++) {
167         var segment = segmentDataServer[i];
168         
169         segmentData[segment.id] = segment;
170     }
171     
172     
173     var segmentWidgets = [];
174     var segmentWidgetNodes = [];
175     
176     
177     /* For each segment data object */
178     for(i = 0, il = segmentDataServer.length; i < il; i++) {
179         var segment = segmentData[segmentDataServer[i].id];
180         
181         /* Connect with actual audio object */
182         var segment_audio_id = segment.audio;
183         var segment_audio_object = audioData[segment_audio_id];
184         /* Tell audio object about this segment */
185         segment_audio_object.segments.push(segment);
186         /* Now audio attribute will point to actual audioData object instead of
187             just the id */
188         segment.audio = segment_audio_object;
189         
190         
191         /* Create segment widget */
192         var segmentWidget = new SegmentWidget({
193             template: segmentWidgetTemplate,
194             context: segment,
195             panel: this
196         });
197         
198         /* Save actual widget */
199         segmentWidgets[segment.id] = segmentWidget;
200         /* Save node for this widget */
201         segmentWidgetNodes.push(segmentWidget.container.get(0));
202         
203     }
204     this.segmentWidgets = segmentWidgets;
205     this.segmentWidgetNodes = segmentWidgetNodes;
206     
207     
208     /* We will store our fileWidgets here indexed by audio id */
209     var fileWidgets = [];
210     
211     /* We will store all of our fileWidget nodes here indexed by their order on 
212         the DOM */
213     var fileWidgetNodes = [];
214     
215     
216     /* For each audio data object */
217     for(i = 0, il = audioDataServer.length; i < il; i++) {
218         var audio = audioData[audioDataServer[i].id];
219         
220         /* Create file widget */
221         var fileWidget = new FileWidget({
222             template: fileWidgetTemplate, 
223             context: audio, 
224             panel: this, 
225         });
226         
227         /* Save widget by audio_id */
228         fileWidget[audio.id] = fileWidget;
229         
230         /* Save node for this widget in list */
231         fileWidgetNodes.push(fileWidget.container.get(0));
232     }
233     
234     this.fileWidgets = fileWidgets;
235     this.fileWidgetNodes = fileWidgetNodes;
236     
237     
238     this.audioData = audioData;
239     this.segmentData = segmentData;
240     
241     /* Default is segments */
242     this.showSegments();
243 };
244 
245 /**
246  *  Replace all nodes in contents container with fileWidgetNodes.
247  **/
248 AudioListPanel.prototype.showFiles = function() {
249     domElementsReplace(this.fileWidgetNodes, this.contents);
250     this.mode = 'files';
251     this.audioSwitcherFilesButton.container.hide();
252     this.audioSwitcherSegmentsButton.container.show();
253     this.segmentsHeader.hide();
254     this.filesHeader.show();
255 };
256 
257 /**
258  *  Replace all nodes in contents container with segmentWidgetNodes.
259  **/
260 AudioListPanel.prototype.showSegments = function() {
261     domElementsReplace(this.segmentWidgetNodes, this.contents);
262     this.mode = 'segments';
263     this.audioSwitcherSegmentsButton.container.hide();
264     this.audioSwitcherFilesButton.container.show();
265     this.segmentsHeader.show();
266     this.filesHeader.hide();
267 };
