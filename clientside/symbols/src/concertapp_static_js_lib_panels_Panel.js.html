  6 /**
  7  *  Any panel that distinguishes groups of functionality on the UI.  This could be
  8  *  the top "global options" bar, or the waveform playback panel.  Panels will
  9  *  contain widgets, which may be buttons or groups of buttons.
 10  *  Panel is meant to be an abstract class.  It doesn't do much on its own.
 11  *  @class
 12  **/
 13 function Panel(params) {
 14     if(params) {
 15         this.init(params);
 16     }
 17 }
 18 
 19 /**
 20  *  Initialize the UI Panel.  Should be called from child class.
 21  *
 22  *  @param  params.container        jQuery object - container for panel  
 23  *  @param  params.page             Page - the page that this panel belongs to
 24  *  @param  params.loading          Boolean -   Wether or not this panel is initially
 25  *                                      to display a loading notification.
 26  **/
 27 Panel.prototype.init = function(params) {
 28     
 29     var page = params.page;
 30     if(typeof(page) == 'undefined') {
 31         throw new Error('params.page is undefined');
 32     }
 33     this.page = page;
 34     
 35     var container = params.container;
 36     if(typeof(container) == 'undefined') {
 37         throw new Error('params.container is undefined for '+container.selector);
 38     }
 39     else if(container.length == 0) {
 40         throw new Error('container not found: '+container.selector);
 41     }
 42     this.container = container;
 43     
 44     /* This is a reference to the panel's contents */
 45     var contents = container.children('.panel_contents');
 46     if(typeof(contents) == 'undefined') {
 47         throw new Error('contents is undefined for panel: '+container.attr('id'));
 48     }
 49     else if(contents.length == 0) {
 50         throw new Error('malformed HTML: contents not found for '+container.attr('id'));
 51     }
 52     this.contents = contents;
 53     
 54     /* This is a reference to the div that contains the panel's header */
 55     var header = container.children('.panel_header');
 56     if(typeof(header) == 'undefined') {
 57         throw new Error('header is undefined for panel '+container.attr('id'));
 58     }
 59     else if(header.length == 0) {
 60         throw new Error('header not found for panel '+container.attr('id'));
 61     }
 62     this.header = header;
 63 
 64     
 65     /* Get the loader element for this panel */
 66     var loader = container.children('.panel_loader');
 67     if(typeof(loader) == 'undefined' || loader.length == 0) {
 68         throw new Error('Malformed HTML.  No panel_loader element found for panel: '+container.attr('id'));
 69     }
 70     this.loader = loader;
 71     
 72     /** This boolean will keep track of if we are loading or not */
 73     var loading = params.loading;
 74     if(typeof(loading) == 'undefined') {
 75         loading = false;
 76     }
 77     this.loading = loading;
 78 
 79     /* If we should be loading, show the loading notification */
 80     if(loading) {
 81         this.showLoadingNotification();
 82     }
 83 }
 84 
 85 /**
 86  *  This function should be called when the panel is loading, or when the panel
 87  *  is done displaying a loading notification.
 88  **/
 89 Panel.prototype.toggleLoadingNotification = function() {
 90     var loading = this.loading;
 91     if(!loading) {
 92         /* Enable loading notification */
 93         this.showLoadingNotification();
 94         this.loading = true;
 95     }
 96     else {
 97         this.hideLoadingNotification();
 98         this.loading = false;
 99     }
100 };
101 
102 /**
103  *  Display the loading notification on this panel.
104  **/
105 Panel.prototype.showLoadingNotification = function() {
106     this.loader.addClass('panel_loader_enabled');    
107 };
108 
109 /**
110  *  Hide the loading notification on this panel.
111  **/
112 Panel.prototype.hideLoadingNotification = function() {
113     this.loader.removeClass('panel_loader_enabled');
114 };
