  7 /**
  8  *  This is a class for the large buttons (right now found in top options bar).  
  9  *  Label appears underneath icon when hovered, and goes away afterwards.
 10  *	@class
 11  **/
 12 function LargeIconButton(params) {
 13     if(params) {
 14         this.init(params);
 15     }
 16 }
 17 LargeIconButton.prototype = new IconButton();
 18 
 19 LargeIconButton.prototype.init = function(params) {
 20     IconButton.prototype.init.call(this, params);
 21 
 22     /* There are no new params. */
 23 
 24 }
 25 
 26 /**
 27  *  We will override the hoverIn so we can show the label (if one exists).
 28  **/
 29 LargeIconButton.prototype.hoverIn = function() {
 30     var label = this.label;
 31     /* If we have a label */
 32     if(typeof(label) != 'undefined') {
 33         /* Add the 'hover' class to our label */
 34         label.addClass('hover');
 35     }
 36 
 37     /* Then call parent */
 38     IconButton.prototype.hoverIn.call(this);
 39 }
 40 
 41 /**
 42  *  We will also override the hoverOut so we can hide the label
 43  **/
 44 LargeIconButton.prototype.hoverOut = function() {
 45     var label = this.label;
 46     
 47     /* If we have a label */
 48     if(typeof(label) != 'undefined') {
 49         /* remove the 'hover' class from the label */
 50         label.removeClass('hover');
 51     }
 52     
 53     /* Then call parent */
 54     IconButton.prototype.hoverOut.call(this);
 55 }
