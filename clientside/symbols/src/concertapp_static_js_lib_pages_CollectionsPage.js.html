  7 /**
  8  *  Initialize all stuff needed on the collections page.
  9  *	@class
 10  *  @extends LoggedInPage
 11  **/
 12 function CollectionsPage(params) {
 13     if(params) {
 14         this.init(params);
 15     }
 16 }
 17 CollectionsPage.prototype = new LoggedInPage();
 18 
 19 CollectionsPage.prototype.init = function(params) {
 20     LoggedInPage.prototype.init.call(this, params);
 21 
 22     /**
 23      *  Create "create/join collection panel"
 24      **/
 25     var createJoinCollectionPanel = new CreateJoinCollectionPanel({
 26         page: this, 
 27         container: $('#create_join_panel'), 
 28         inputElement: $('#create_join_input'), 
 29         resultsElement: $('#create_join_results'),
 30         resultTemplate: $('#create_join_result'),
 31         createNewTemplate: $('#create_join_create_new')
 32     });
 33     this.createJoinCollectionPanel = createJoinCollectionPanel;
 34     
 35     /**
 36      *  "ManageCollectionsPanel"
 37      **/
 38     var manageCollectionsPanel = new ManageCollectionsPanel({
 39         page: this, 
 40         container: $('#manage_collections_panel'), 
 41         collectionTemplate: $('#collection_template'),
 42     });
 43     this.manageCollectionsPanel = manageCollectionsPanel;
 44     
 45     /** Connect panels **/
 46     createJoinCollectionPanel.manageCollectionsPanel = manageCollectionsPanel;
 47     manageCollectionsPanel.globalOptionsPanel = this.globalOptionsPanel;
 48     
 49     /* Retrieve collections to manage */
 50     manageCollectionsPanel.retrieveAndUpdateCollections();
 51     
 52 }
