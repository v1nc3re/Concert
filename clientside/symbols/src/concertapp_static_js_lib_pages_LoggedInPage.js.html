  7 /**
  8  *  This class will run certain functionality that must be run on every page where
  9  *  a user is logged in.
 10  *	@class
 11  **/
 12 function LoggedInPage(params) {
 13     if(params) {
 14         this.init(params);
 15     }
 16 }
 17 LoggedInPage.prototype = new Page();
 18 
 19 LoggedInPage.prototype.init = function(params) {
 20     Page.prototype.init.call(this, params);
 21     
 22     /* Get data from the server for this page */
 23     var data = params.data;
 24     if(typeof(data) == 'undefined') {
 25         throw new Error('params.data is undefined');
 26     }
 27     else if(data.length == 0) {
 28         throw new Error('data not found');
 29     }
 30     this.data = data;
 31     
 32     /* Every page needs the collection data */
 33     var collectionData = data.collections;
 34     if(typeof(collectionData) == 'undefined') {
 35         throw new Error('data.collections is undefined');
 36     }
 37     
 38     /* Now lets create our Backbone collection */
 39     var collections = new Collections;
 40     
 41     /* And populate it with all of the data from the server */
 42     collections.refresh(collectionData);
 43     
 44     /* Save Backbone collection of Concert Collections */
 45     this.collections = collections;
 46     
 47     
 48 
 49     /*  Create the globalOptionsPanel (the buttons and menus at the top of every 
 50         page) */
 51     this.globalOptionsPanel = new GlobalOptionsPanel({
 52         page: this, 
 53         container: $('#global_options_panel')
 54     });
 55 }
