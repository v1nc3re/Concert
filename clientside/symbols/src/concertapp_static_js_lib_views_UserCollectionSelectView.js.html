  7 var UserCollectionSelectView = Backbone.View.extend({
  8     tagName: 'select',
  9     className: 'collection_selector', 
 10     
 11     initialize: function() {
 12         
 13         var params = this.options;
 14         
 15         var collectionSelectorOptionsTemplate = params.collectionSelectorOptionsTemplate;
 16         if(typeof(collectionSelectorOptionsTemplate) == 'undefined') {
 17             throw new Error('params.collectionSelectorOptionsTemplate is undefined');
 18         }
 19         else if(collectionSelectorOptionsTemplate.length == 0) {
 20             throw new Error('collectionSelectorOptionsTemplate not found');
 21         }
 22         this.collectionSelectorOptionsTemplate = collectionSelectorOptionsTemplate;
 23         
 24         this.collectionSelector = params.collectionSelector;
 25         
 26         _.bindAll(this, "render");
 27     },
 28     
 29     render: function() {
 30 
 31         /* Populate dropdown */
 32         this.collectionSelector.html(
 33             this.collectionSelectorOptionsTemplate.tmpl({
 34                 collections: this.collection.toJSON(), 
 35             })
 36         );
 37         
 38         
 39         return this;
 40     }
 41 });
