  7 /**
  8  *  Panel that lists collections for the user to manage.
  9  *	@class
 10  **/
 11 function ManageCollectionsPanel(params) {
 12     if(params) {
 13         this.init(params);
 14     }
 15 }
 16 ManageCollectionsPanel.prototype = new Panel();
 17 
 18 /**
 19  *  @param  params.collectionTemplate        jQuery tmpl - for the collection row.  
 20  **/
 21 ManageCollectionsPanel.prototype.init = function(params) {
 22     Panel.prototype.init.call(this, params);
 23 
 24     /* Template for each collection in the list */
 25     var collectionTemplate = params.collectionTemplate;
 26     if(typeof(collectionTemplate) == 'undefined') {
 27         throw new Error('params.collectionTemplate is undefined');
 28     }
 29     else if(collectionTemplate.length == 0) {
 30         throw new Error('collectionTemplate not found');
 31     }
 32     this.collectionTemplate = collectionTemplate;
 33 
 34     /* get the "table" where we will display the collections */
 35     var collectionsTable = $(this.contents).children('#manage_collections_table');
 36     if(typeof(collectionsTable) == 'undefined') {
 37         throw new Error('collectionsTable is undefined');
 38     }
 39     else if(collectionsTable.length == 0) {
 40         throw new Error('malformed HTML: collectionsTable not found');
 41     }
 42     this.collectionsTable = collectionsTable;
 43     
 44     var collectionsTableHeader = $(this.contents).find('#manage_collections_table_header');
 45     if(typeof(collectionsTableHeader) == 'undefined') {
 46         throw new Error('collectionsTableHeader is undefined');
 47     }
 48     else if(collectionsTableHeader.length == 0) {
 49         throw new Error('malformed HTML: collectionsTableHeader not found');
 50     }
 51     this.collectionsTableHeader = collectionsTableHeader;
 52 
 53     
 54 
 55     
 56     
 57     
 58     /* reference to globalOptionsPanel so we can call methods on there */
 59     this.globalOptionsPanel = null;    
 60     
 61     /* Keep all the collection widgets in case we need them for something.  Will be
 62         indexed by collection_id */
 63     this.collections = [];
 64     
 65 }
 66 
 67 /**
 68  *  Retrieves collections from backend, then updates the list.
 69  **/
 70 ManageCollectionsPanel.prototype.retrieveAndUpdateCollections = function() {
 71     /* Loading */
 72     this.toggleLoadingNotification();
 73     
 74     /* Retrieve JSON data for manage collections list */
 75     $.getJSON('user/', function(me) {
 76         return function(data, status) {
 77             me.updateCollections(data);
 78             /* un loading */
 79             me.toggleLoadingNotification();
 80         };
 81     }(this));
 82 }
 83 
 84 /**
 85  *  Updates the collection list.
 86  **/
 87 ManageCollectionsPanel.prototype.updateCollections = function(data) {
 88     
 89     var collectionTemplate = this.collectionTemplate;
 90     var collectionsTable = this.collectionsTable;
 91     
 92     var collections = [];
 93     
 94     /* Temporary fragment for all DOM additions */
 95     var frag = document.createDocumentFragment();
 96     
 97     /* The header will always be first in the table */
 98     frag.appendChild(this.collectionsTableHeader.get(0))
 99 
100     for(i = 0, il = data.length; i < il; i++) {
101         var col = data[i];
102         
103         var widget = new ManageCollectionWidget({
104             template: collectionTemplate, 
105             context: col,
106             panel: this
107         });
108         
109         collections[col.id] = widget;
110         
111         frag.appendChild(widget.container.get(0));
112     
113     }
114     
115     /* Update manage collections list */
116     collectionsTable.empty().append(frag);
117     
118     this.collections = collections;
119     
120     /* Update dropdown */
121 //    this.globalOptionsPanel.updateCollectionSelector(data);
122 }
123 
124 /**
125  *  This method deletes a collection from the manage collections panel, then
126  *  informs the global options panel that we should remove the collection from there.
127  *
128  *  @param  collection_id        Number  -  The id of the collection to delete.
129  **/
130 ManageCollectionsPanel.prototype.deleteCollection = function(collection_id) {
131     var collection = this.collections[collection_id];
132     
133     /* Remove row from manage table */
134     collection.container.remove();
135     /* Remove reference to self from manageCollectionsPanel */
136     delete this.collections[collection_id];
137     
138     /* Tell global options panel to delete collection */
139     this.globalOptionsPanel.removeCollectionFromSelector(collection_id);
140 };
