1 /**
  2  *  @file       init.js
  3  *  This file contains the function that is called from the bottom of the template,
  4  *  to initialize all client side code.  It also contains the individual functions
  5  *  that are used to initialize each page on the UI.
  6  *  @author     Colin Sullivan <colinsul [at] gmail.com>
  7  **/
  8 
  9 /**
 10  *  Initialize all client side functionality by creating the applicable
 11  *  page object.
 12  *
 13  *  @param  pagePath        String  -   From the server, the string that we will
 14  *  use to determine what page we are on.  This variable is sent in on 
 15  *  base_site.html template.
 16  *  @param  data        Object   -   the JSON object that contains all of the
 17  *  data we need for this page.
 18  **/
 19 function initializeUI(pagePath, data) {
 20     $(document).ready(function(){
 21         /* Make all fields in the future that have the "autoClear" class on them    
 22             autoclear */
 23         initializeAutoClearFieldBehavior();
 24 
 25         /* Make global notifier object that we can use anywhere to notify the user */
 26         com.concertsoundorganizer.notifier = new Notifier({});
 27 
 28         /* image urls put here will be loaded 
 29         preloadImages([
 30             '/graphics/ajax-loader.gif',
 31             '/graphics/somethingelse.jpg'
 32         ]);
 33         */    
 34         
 35         var pageParams = {
 36             data: data, 
 37         };
 38 
 39         /* For each page, run JS corresponding to that page */
 40         var pageInitializers = {
 41             '/login/': LoginPage,
 42             '/dashboard/': DashboardPage,
 43             '/collections/': CollectionsPage,
 44             '/audio/upload/': UploadPage,
 45             '/organize/collection/': OrganizePage
 46         };
 47 
 48         /* Run the initializer function for this page. */
 49         var page = new pageInitializers[pagePath](pageParams);
 50     });
 51 }
